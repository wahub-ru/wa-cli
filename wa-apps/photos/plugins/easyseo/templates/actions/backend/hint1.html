{literal} 
  
Метаданные

  {$store_info.name} - название магазина
  {$store_info.phone} - телефон магазина
  {$storefront.name} - название витрины
  {$host} - текущий домен
  {$wa_app_url} – ссылка на витрину


  
Страницы

  {$page.name} - название страницы
  {$page.title} - заголовок страницы
  {$page.keywords} - ключевые слова страницы

  
Альбомы

  {$album.name} - название альбома
  {$album.note} - заметка по альбому
  {$album.description} - заметка по альбому
  {$photos|count} - количество фото
  {$photos} - список фотографий

  
Пагинация

  {$page_number} - номер страницы
  {$pages_count} - количество страниц

  
Фото

  {$photo.name} - название фото
  {$photo.description} - описание фото
  {$photo.rate} - рейтинг фото
  {$photo.width} - ширина фото
  {$photo.height} - высота фото
  {$photo.url} - адрес фото
  {$photo.prev_photo_url} - ссылка на предыдущее фото
  {$photo.next_photo_url} - ссылка на следующее фото
  {$albums|count} - количество альбомов
  {$albums} - список альбомов
{/literal}
